# Listing 3.11 — idempotent_post.txt: a write request with an idempotency key, and its replay

# First attempt. The consumer generates a key and includes it as a header.
# The server processes the request and records the key.
POST /orders HTTP/1.1
Content-Type: application/json
Idempotency-Key: 9f3e2c91-1a4b-4d7e-8c0f-2b5a6e9d0f1a       # A — UUID; fresh per new operation, reused on retries

{ "customer_id": "cus_17", "items": [{ "sku": "p_001", "qty": 1 }] }

HTTP/1.1 201 Created
{ "id": "ord_42", "status": "PENDING", "total_cents": 8900 }

---

# Second attempt — same key. The consumer did not receive the first response
# and retries. The server recognises the key and returns the original response
# without creating a duplicate order.
POST /orders HTTP/1.1
Idempotency-Key: 9f3e2c91-1a4b-4d7e-8c0f-2b5a6e9d0f1a       # B — server compares key only, not body

HTTP/1.1 201 Created
Idempotent-Replay: true                                        # C — signals replay from store; consumer logs and moves on
{ "id": "ord_42", "status": "PENDING", "total_cents": 8900 }
